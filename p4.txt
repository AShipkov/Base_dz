import random
a = 1
b = 100
number = random.randint (a,b)
while True :
    print(('Ваше число'), (number), ('?'))
    user = input()
    if user == '=' :
        print('Победа')
        break
    elif user == '<' :
        b = number-1
        number = random.randint(a, b)
    else :
        a = number+1
        number = random.randint(a, b)
